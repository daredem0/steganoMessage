133int Png::readPng_init(FILE *inFile, uint32_t *pWidth, uint32_t *pHeight, int *bitDepth, int *colorType){
134 //check again if the file really is a png file
135 unsigned char signature[8]; //8 bits are identical for every png file
136 fread(signature, 1, 8, inFile);
137 if(!png_check_sig(signature, 8))
138 return ErrorHandler::errPngSig;
139
140 /*Build image information structs and check for out of memory errors*/
141 /*png_structp is used internally to keep track of the current state of the png image*/
142 png_ptr = png_create_read_struct(PNG_LIBPNG_VER_STRING, NULL, NULL, NULL);
143 if(!png_ptr)
144 return ErrorHandler::errPngMem;
145
146 /*png_infop is used to indicate what the state of the pnc will be after user transformations are performed*/
147 info_ptr = png_create_info_struct(png_ptr);
148 if(!info_ptr){
149 png_destroy_read_struct(&png_ptr, NULL, NULL);
150 return ErrorHandler::errPngMem;
151 }
152
153 //libpng specific error handling
154 if(setjmp(png_jmpbuf(png_ptr))){
155 png_destroy_read_struct(&png_ptr, &info_ptr, NULL);
156 return ErrorHandler::errPngInt;
157 }
158
159 //do some init stuff
160 png_init_io(png_ptr, inFile);
161 png_set_sig_bytes(png_ptr, 8);
162
163 //now we read basically everything until the image data
164 png_read_info(png_ptr, info_ptr);
165
166 /*Lastly we write the important information from the header to the addresses where the pointer arguments point*/
167 png_get_IHDR(png_ptr, info_ptr, (png_uint_32*)pWidth, (png_uint_32*)pHeight, bitDepth, colorType, NULL, NULL, NULL);
168 return ErrorHandler::errNoError;

171int Png::readpng_get_bgcolor(uint8_t *red, uint8_t *green, uint8_t *blue){
172 //libpng specific error handling
173 if(setjmp(png_jmpbuf(png_ptr))){
174 png_destroy_read_struct(&png_ptr, &info_ptr, NULL);
175 return ErrorHandler::errPngInt;
176 }
177
178 //check if png file has backgroundcolor set
179 if(!png_get_valid(png_ptr, info_ptr, PNG_INFO_bKGD)){
180 bKgdFlag = false;
181 }
182
183 //do this only if the backgroundcolor is set
184 if(bKgdFlag == true){
185 //finally get the backgroundcolor from libpng
186 png_color_16p pBackground;
187 png_get_bKGD(png_ptr, info_ptr, &pBackground);
188
189 //convert to 8bit values
190 if(bDepth == 16){
191 *red = pBackground->red >> 8;
192 *green = pBackground->green >> 8;
193 *blue = pBackground->blue >> 8;
194 }
195 else if(cType == PNG_COLOR_TYPE_GRAY && bDepth < 8){
196 switch(bDepth){
197 case 1:
198 *red = *green = *blue = pBackground->gray? 255 : 0;
199 break;
200 case 2:
201 *red = *green = *blue = (255/3) * pBackground->gray;
202 break;
203 default: /* means bDepth = 4*/
204 *red = *green = *blue = (255/15) * pBackground->gray;
205 }
206 }
207 else{
208 *red = pBackground->red;
209 *green = pBackground->green;
210 *blue = pBackground->blue;
211 }
212 }
213 else{ /*if no background color is set in the image make it white*/
214 errHandle->printLog("No background color found\n");
215 *red = *green = *blue = 255;
216 }
217
218 return ErrorHandler::errNoError;

221int Png::readpng_get_image(double display_exponent, int *pChannels, unsigned long *pRowBytes){
222 //libpng specific error handling
223 if(setjmp(png_jmpbuf(png_ptr))){
224 png_destroy_read_struct(&png_ptr, &info_ptr, NULL);
225 return ErrorHandler::errPngInt;
226 }
227 //set up png to give us 8 bit per channel + alpha if available.
228 /*png_set_expand() shall set transformation in png_ptr such that paletted
229 * images are expanded to RGB, grayscale images of bit-depth less than 8 are expanded to
230 * 8-bit images and tRNS chunks are expanded to alpha channels.*/
231 if (cType == PNG_COLOR_TYPE_PALETTE)
232 png_set_expand(png_ptr);
233 if (cType == PNG_COLOR_TYPE_GRAY && bDepth < 8)
234 png_set_expand(png_ptr);
235 if (png_get_valid(png_ptr, info_ptr, PNG_INFO_tRNS))
236 png_set_expand(png_ptr);
237 if (bDepth == 16)
238 png_set_strip_16(png_ptr);
239 if (bDepth == PNG_COLOR_TYPE_GRAY ||
240 bDepth == PNG_COLOR_TYPE_GRAY_ALPHA)
241 png_set_gray_to_rgb(png_ptr);
242
243 //apply not existing gamma correction
244 double gamma;
245 if (png_get_gAMA(png_ptr, info_ptr, &gamma))
246 png_set_gamma(png_ptr, display_exponent, gamma);
247
248 //update information struct after transformations are done and get values for the pointers
249 png_uint_32 rowbytes;
250 png_bytep row_pointers[height];
251 //row_pointers = (png_bytep*)malloc(sizeof(png_bytep)*height);
252
253 png_read_update_info(png_ptr, info_ptr);
254
255 *pRowBytes = rowbytes = png_get_rowbytes(png_ptr, info_ptr);
256 *pChannels = (int)png_get_channels(png_ptr, info_ptr);
257
258 //allocate memory, if we cant allocate it return memory error
259 pData = NULL;
260 errHandle->printLog("Allocating memory\n");
261 if((pData = (unsigned char*)malloc(rowbytes*height)) == NULL){
262 png_destroy_read_struct(&png_ptr, &info_ptr, NULL);
263 return ErrorHandler::errPngMem;
264 }
265
266 //store the addresses in the row_pointers array
267 for(png_uint_32 i = 0; i < height; ++i){
268 row_pointers[i] = pData + i * rowbytes;
269 }
270
271 //finally store the image data in the created array
272 png_read_image(png_ptr, row_pointers);
273 //png_read_end(png_ptr, NULL); //not sure if we need that one
274
275 return ErrorHandler::errNoError;
276}
