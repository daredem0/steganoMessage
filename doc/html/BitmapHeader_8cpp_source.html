131int BitmapHeader::read(std::ifstream& f){
132 try{
133 f.seekg(0, std::ios::beg);
134 //File Header
135 f.read((char*) &header.bfType, sizeof(header.bfType));
136 fixType();
137 f.read((char*) &header.bfSize, sizeof(header.bfSize));
138 f.read((char*) &header.bfReserved, sizeof(header.bfReserved));
139 f.read((char*) &header.bfOffBits, sizeof(header.bfOffBits));
140 //Information Header
141 f.read((char*) &header.biSize, sizeof(header.biSize));
142 f.read((char*) &header.biWidth, sizeof(header.biWidth));
143 f.read((char*) &header.biHeight, sizeof(header.biHeight));
144 f.read((char*) &header.biPlanes, sizeof(header.biPlanes));
145 f.read((char*) &header.biBitCount, sizeof(header.biBitCount));
146 f.read((char*) &header.biCompression, sizeof(header.biCompression));
147 f.read((char*) &header.biSizeImage, sizeof(header.biSizeImage));
148 f.read((char*) &header.biXPelsPerMeter, sizeof(header.biXPelsPerMeter));
149 f.read((char*) &header.biYPelsPerMeter, sizeof(header.biYPelsPerMeter));
150 f.read((char*) &header.biClrUsed, sizeof(header.biClrUsed));
151 f.read((char*) &header.biClrImportant, sizeof(header.biClrImportant));
152
153 if(header.bfOffBits > f.tellg()){
154 std::cout << "Building header leftover size: " << header.bfOffBits - f.tellg() << std::endl;
155 f.read((char*)&header.leftover, header.bfOffBits - f.tellg());
156 }
157
158 //fix bitcount
159 std::cout << "Original Bitcount: " << header.biBitCount << std::endl;
160 header.oldBitCount = header.biBitCount;
161 if(header.biBitCount == 24){
162 header.biBitCount = 32;
163 }
164 return errNoError;
165 }
166 catch(const std::exception& e){
167 errHandle->printErrorStdEx(e);
168 errHandle->printError(errBmHeadRead);
169 return errBmHeadRead;
170 }
171 catch(...){
172 errHandle->printError(errUnknown);
173 errHandle->printError(errBmHeadRead);
174 return errBmHeadRead;
175 }
176 return errBmHeadRead;
177}
