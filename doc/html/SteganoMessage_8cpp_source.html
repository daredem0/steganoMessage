45int SteganoMessage::initialize(int argc, char *argv[]){
46 int errTemp;
47 try{
48 //initialization stuff, especially read the header info, else everything will go horribly wrong in all possible ways
49 err->printLog("Trying to get path\n");
50 std::string tempPath;
51 argv[2] == NULL ? tempPath = NOPATH : tempPath = (std::string)argv[2]; //if there was a path given store it in tempPath
52 if(tempPath == NOPATH)
53 throw errPath; //throw error if we didnt get a path
54 err->printLog("Found Path: " + tempPath + "\n");
55 errTemp = this->buildImage(tempPath); //Call constructor for image type object and set path
56 if(errTemp != 0) throw errTemp;
57 err->printLog("Built path: " + this->getImage()->getPath() + "\n");
58 //set filters if activated
59 if(argc > 3){
60 if(argv[3] == FILTER){
61 this->setFilterMode(argv[4]);
62 crazy = false;
63 argc > 5 ? this->setLog(true) : this->setLog(false);
64 }
65 else if(argv[3] == CRFILTER){
66 this->setFilterMode(argv[4]);
67 crazy = true;
68 argc > 5 ? this->setLog(true) : this->setLog(false);
69 }
70 else if(argv[3] == LOG){
71 this->setFilterMode("");
72 this->setLog(true);
73 }
74 }
75 else{
76 this->setFilterMode("");
77 this->setLog(false);
78 }
79 err->printLog("Filter mode: " + std::to_string(stegFilter) + "\n");
80 if(img->identifyFileFormat(this->getImage()->getPath()) == BITMAP)
81 errTemp = this->getImage()->readImage(); //extract the image information
82 else if(img->identifyFileFormat(this->getImage()->getPath()) == JPEG)
83 errTemp = this->getImage()->readImage(); //extract the image information
84 else if(img->identifyFileFormat(this->getImage()->getPath()) == PNG){
85 //errTemp = ((Png*)this->getImage())->read_png_file(this->getImage()->getPath()); //extract the image information
86 errTemp = this->getImage()->readImage();

89 }
